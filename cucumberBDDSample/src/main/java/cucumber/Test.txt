public class HeapInfo {
    public static void main(String[] args) {
        Runtime runtime = Runtime.getRuntime();

        long maxMemory = runtime.maxMemory();       // Maximum heap size (bytes)
        long totalMemory = runtime.totalMemory();   // Current heap size (bytes)
        long freeMemory = runtime.freeMemory();     // Free heap space (bytes)

        System.out.println("Max heap size: " + (maxMemory / (1024 * 1024)) + " MB");
        System.out.println("Total heap size: " + (totalMemory / (1024 * 1024)) + " MB");
        System.out.println("Free heap space: " + (freeMemory / (1024 * 1024)) + " MB");
    }
}
