
SELECT 
    JSON_VALUE(ATTRIBUTE_DOCUMENT, '$.ba360') AS ba360_value, 
    JSON_VALUE(ATTRIBUTE_DOCUMENT, '$.ba360_pending') AS ba360_pending_value, 
    COUNT(*) AS count
FROM 
    MCESSERV.ATTRIBUTE_VALUE
WHERE 
    ATTRIBUTE_PKREF = 40
GROUP BY 
    JSON_VALUE(ATTRIBUTE_DOCUMENT, '$.ba360'), 
    JSON_VALUE(ATTRIBUTE_DOCUMENT, '$.ba360_pending');
