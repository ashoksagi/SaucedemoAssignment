import java.util.ArrayList;
import java.util.Collections;
import java.util.List;

public class RandomCompanySelector {
    public static void main(String[] args) {
        List<String> companies = new ArrayList<>();
        
        // Populate the list with company names (for demonstration purposes)
        companies.add("Company A");
        companies.add("Company B");
        companies.add("Company C");
        // Add more companies to the list
        
        // Shuffle the list to randomize the order
        Collections.shuffle(companies);
        
        // Get the first 50 companies from the shuffled list
        List<String> randomCompanies = companies.subList(0, Math.min(50, companies.size()));
        
        // Print the selected companies
        for (String company : randomCompanies) {
            System.out.println(company);
        }
    }
}
