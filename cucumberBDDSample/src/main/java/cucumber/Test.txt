import io.restassured.RestAssured;
import io.restassured.http.ContentType;
import io.restassured.module.jsv.JsonSchemaValidator;
import org.junit.jupiter.api.Test;

import static io.restassured.RestAssured.*;

public class ApiSchemaValidationTest {

    @Test
    public void validateApiResponseSchema() {
        RestAssured.baseURI = "https://your-api-endpoint.com";

        given()
            .contentType(ContentType.JSON)
        .when()
            .get("/users/123") // Replace with the actual endpoint
        .then()
            .assertThat()
            .body(JsonSchemaValidator.matchesJsonSchemaInClasspath("expectedSchema.json"))
            .statusCode(200);
    }
}
